country_event = {

	title = ww1_ethiopia.5.t
	desc = ww1_ethiopia.5.d
	fire_only_once = yes
	is_triggered_only = yes
	trigger = {
		is_in_faction = no
		is_subject = no
		OR = {
			GER = {
				is_faction_leader = yes
			}
			TUR = {
				AND = {
					has_completed_focus = TUR_support_iyasu
					is_faction_leader = yes
				}
			}
		}
	}
	option = {
		name = ww1_ethiopia.5.a
		if = {
			limit = {
				TUR = {
                    has_completed_focus = TUR_support_iyasu
					is_faction_leader = yes
				}
			}
			TUR = {
				add_to_faction = ETH
			}
			else = {
				GER = {
					add_to_faction = ETH
				}
			}
		}
	}
}
