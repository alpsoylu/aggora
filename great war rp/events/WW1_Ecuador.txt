country_event = {
	id = ww1_ecuador.2
	title = ww1_ecuador.2.t
	desc = {
		text = ww1_ecuador.2.d
		trigger = {
			has_country_flag = Emilio_Estrada_chosen
		}
	}
	desc = {
		text = ww1_ecuador.2.d_2
		trigger = {
			has_country_flag = Flavio_Alfaro_chosen
		}
	}
	picture = GFX_report_event_election_vote
	fire_only_once = yes
	mean_time_to_happen = {
		days = 25
	}
	trigger = {
		original_tag = ECU
		OR = {
			has_country_flag = Flavio_Alfaro_chosen
			has_country_flag = Emilio_Estrada_chosen
		}
		has_country_leader = {
			name = "Eloy Alfaro"
			ruling_only = yes
		}
		date > 1911.6.5
	}
	option = {
		name = ww1_ecuador.2.a
		set_country_flag = Eloy_Alfaro_resigns
		if = {
			limit = {
				has_country_flag = Flavio_Alfaro_chosen
			}
			create_country_leader = {
				name = "Flavio Alfaro"
				picture = "gfx/leaders/ECU/ECU_flavio_alfaro.dds"
				expire = "1935.1.1"
				ideology = social_liberalism
				traits = {
					trait_social_liberalism
				}
			}
		}
		if = {
			limit = {
				has_country_flag = Emilio_Estrada_chosen
			}
			create_country_leader = {
				name = "Emilio Estrada"
				picture = "gfx/leaders/ECU/ECU_emilio_estrada.dds"
				expire = "1935.1.1"
				ideology = classical_liberalism
				traits = {
					trait_classical_liberalism
				}
			}
			add_political_power = -50
		}
		ai_chance = {
			factor = 30
			modifier = {
				is_historical_focus_on = yes
				factor = 0
			}
		}
	}
	# I am the Senate
	option = {
		name = ww1_ecuador.2.b
		trigger = {
			has_country_flag = Emilio_Estrada_chosen
		}
		add_stability = -0.1
		add_popularity = {
			ideology = neutrality
			popularity = 0.15
		}
		set_country_flag = Eloy_Alfaro_keept_the_power
		ai_chance = {
			factor = 70
		}
		hidden_effect = {
			country_event = {
				id = ww1_ecuador.3
				days = 30
				random = 5
			}
		}
	}
}
