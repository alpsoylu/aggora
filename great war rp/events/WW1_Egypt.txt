			has_country_flag = great_war_loser
		}
		TUR = {
			owns_state = 454

		}
	}
	mean_time_to_happen = { days = 30 }

	option = {
		name = ww1_egypt.5.a

		hidden_effect = {
			TUR = {
				if = {
					limit = {
						has_country_flag = great_war_winner
					}

						id = ww1_egypt.7
						random = 3
					}
				}
			}
		}

		ai_chance = {
			base = 100
			modifier = {
				factor = 0

					value > 50
				}
			}
		}
	}
	option = {
		name = ww1_egypt.5.b

			base = 0
			modifier = {
