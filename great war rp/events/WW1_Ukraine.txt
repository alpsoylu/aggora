#FALL OF HETMANATE
###################
country_event = {
	id = ww1_ukraine.4
	title = ww1_ukraine.4.t
	desc = ww1_ukraine.4.d
	fire_only_once = yes
	trigger = {
		tag = UKR
		has_global_flag = great_war_ends
		has_cosmetic_tag = UKR_GETMAN
		has_country_flag = ukrainian_hetmanate
		GER = {
			OR = {
				has_country_flag = great_war_loser
				is_subject = yes
				exists = no
				has_civil_war = yes
				has_country_flag = third_franco_german_war_loser
			}
		}
	}
	mean_time_to_happen = {
		days = 30
	}
	option = {
		name = ww1_ukraine.4.a
		GER = { end_puppet = UKR}
		if = {
			limit = {
				is_in_faction_with = GER
			}
			leave_faction = yes
		}
		create_country_leader = {
			name = "Volodymyr Vynnychenko"
			picture = "gfx/leaders/UKR/UKR_vladimir_vynnychenko.dds"
			expire = "1935.1.1"
			ideology = socialism
			traits = {
				trait_socialism
			}
		}
		set_politics = {
			ruling_party = democratic
			elections_allowed = yes
		}
		set_popularities = {
			democratic = 50
			fascism = 0
			communism = 40
			neutrality = 10
		}
		set_party_name = {
			ideology = democratic
			long_name = UKR_directory_party							
			name = UKR_directory_party
    	}
    	drop_cosmetic_tag = yes
    	clr_country_flag = ukrainian_hetmanate
		clr_country_flag = protectorate
    	ai_chance = { 
    		factor = 100
    	}
    }
}
