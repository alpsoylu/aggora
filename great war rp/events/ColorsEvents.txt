add_namespace = color

# Soviet-Red Color
####################
country_event = {
	id = color.1
	
	trigger = {
		is_puppet_of = SOV
		has_government = communism
		NOT = { 
			has_cosmetic_tag = SOV_puppet 
			tag = POL
			tag = GER
			tag = FRA
			tag = ITA
			tag = FIN
			has_cosmetic_tag = ITA_socialism
			has_cosmetic_tag = SRI
			has_cosmetic_tag = GEO_TSCN_commissariat
			has_cosmetic_tag = GEO_TSCN
			has_cosmetic_tag = TUR_REP
			has_cosmetic_tag = TUR_SOC
			has_cosmetic_tag = TUR_TRN
		}
		is_major = no
	}	
	mean_time_to_happen = {
		days = 0
	}

	hidden = yes

	immediate = {
		set_cosmetic_tag = SOV_puppet
	}
}

# Wallachia
####################
country_event = {
	id = color.10
	
	trigger = {
		original_tag = ROM
		MOL = { exists = yes }
		NOT = {
		    any_state = {
				is_core_of = MOL
				NOT = { state = 80 }
                is_owned_by = ROM
            } 
		}
        NOT = {
            AND = {		
                TRA = { exists = no }
				any_state = {
					is_core_of = TRA
					NOT = { state = 80 }
                    is_owned_by = ROM
                }
            }
        }
        NOT = { has_cosmetic_tag = WAL }		
        has_civil_war = no
	}	
	mean_time_to_happen = {
		days = 0
	}

	hidden = yes

	immediate = {
		set_cosmetic_tag = WAL
	}
}
