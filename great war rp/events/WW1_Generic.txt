	id = ww1_generic.3
	title = ww1_generic.3.t
	desc = ww1_generic.3.d

	trigger = {
		has_war = no
		num_of_factories > 10
		NOT = {
			has_government = fascism
		}
		NOT = {
			has_country_flag = crisis_in_progress
		}
		NOT = {
			has_country_flag = workers_protest
		}
		if = {
			limit = {
				has_government = democratic
			}

		}
		if = {
			limit = {
				has_government = neutrality
			}
			neutrality < 0.5
		}
		if = {
			limit = {
				has_government = communism
			}

		}

		set_country_flag = workers_protest

		days = 720

			factor = 4

		}

			factor = 1.5

		}

			factor = 1.5

		}

			factor = 0.75
			has_idea = partial_economic_mobilisation
		}
		modifier = {
			factor = 0.75
			has_idea = war_economy
		}
		modifier = {
			factor = 2
			has_government = communism
		}

			factor = 0.75

				has_government = communism

			}

				has_government = democratic
			}

				ideology = communism
				popularity = 0.02
			}
		}

			limit = {
				has_government = communism
			}
			add_popularity = {
				ideology = democratic
				popularity = 0.02
			}
		}
		clr_country_flag = workers_protest
		ai_chance = {
			factor = 50
		}
	}
	option = {
		name = ww1_generic.3.b
		random_list = {
			50 = {
				modifier = {
					factor = 0.75
					has_war_support < 0.5
				}
				country_event = ww1_generic.4				#забастовка разогнана
			}
			50 = {
				country_event = ww1_generic.5				#кровавое воскресенье
			}

				factor = 10
				has_government = neutrality
			}

				factor = 10
				has_government = communism
			}
		}
	}

		name = ww1_generic.3.c
		if = {
			limit = {
				has_country_flag = labor_reform

		}
