add_namespace = ww1_estonia

# Collapse of Baltic Duchy
####################
country_event = {
	id = ww1_estonia.1
	trigger = {
		GER = { has_country_flag = great_war_loser }
		NOT = { has_global_flag = versallies_treaty}
		OR = {
			GER = {
				AND = {
				    owns_state = 13
				    owns_state = 191
				    owns_state = 12			
				}
			}	
			UBD = { exists = yes }
        }
	}
	fire_only_once = yes
	hidden = yes
	mean_time_to_happen = { days = 31 }
	immediate = {
		hidden_effect = {
			if = {
				limit = {
					LAT = {
						OR = {
					        has_country_flag = LAT_iskolata
						    has_cosmetic_tag = LAT_iskolata
						}
						exists = no
					}
                }					
			    LAT = {
				    clr_country_flag = LAT_iskolata	
				    drop_cosmetic_tag = yes
					retire_ideology_leader = communism
					set_politics = {
					   	ruling_party = democratic
						elections_allowed = yes
					}
					set_rule = {
				        can_join_factions = yes
				    }
					set_popularities = {
					   	democratic = 50
					   	neutrality = 30
					   	fascism = 0
					   	communism = 20
				    }
				}
			}	
		    every_state = {
			    limit = {
				    OR = {
					   is_core_of = LIT 
					   is_core_of = LAT
					   is_core_of = EST
				    }
				    NOT = { state = 96 }
			   }
			   remove_core_of = RUS
			   remove_core_of = SOV
		    }		     
			SOV = { transfer_state = 96 }
		}
		set_global_flag = estonia_released
		EST = {
			transfer_state = 13
			transfer_state = 191
		}
		set_global_flag = latvia_released
        LAT = {
			transfer_state = 12
			transfer_state = 96
			transfer_state = 190
		}
		every_state = {
		    limit = {	
			    is_core_of = UBD
			}
			remove_core_of = UBD
		}
	}
}
