#Belgium rebels against Germany
country_event = {
	id = ww1_belgium.6
	title = ww1_belgium.6.t
	desc = ww1_belgium.6.d
	trigger = {
		tag = BEL
		num_divisions > 1
		NOT = {
			has_government = democratic
		}
		is_subject_of = GER
		has_idea = BEL_belgian_resistance
		democratic > 0.4
		GER = {
			has_war = yes
		}
	}
	fire_only_once = yes
	immediate = {
		hidden_effect = {
			GER = {
				country_event = {
					id = ww1_belgium.7
				}
			}
			start_civil_war = {
				ruling_party = neutrality
				ideology = democratic
				size = 0.5
			}
			random_other_country = {
				limit = {
					original_tag = BEL
					has_government = democratic
				}
				create_field_marshal = {
					name = "Albert I"
					portrait_path = "gfx/leaders/BEL/BEL_albert_general.dds"
					traits = {
						offensive_doctrine
						guerilla_fighter
					}
					skill = 5
				}
				drop_cosmetic_tag = yes
				hidden_effect = {
					set_party_name = {
						ideology = neutrality
						long_name = BEL_neutrality_party_long
						name = BEL_neutrality_party
					}
					set_party_name = {
						ideology = democratic
						long_name = BEL_resistance_long
						name = BEL_resistance
					}
					create_country_leader = {
						name = "Albert I"
						picture = "gfx/leaders/BEL/BEL_albert_general.dds"
						expire = "1935.1.1"
						ideology = conservatism
						traits = {
							trait_conservatism
						}
					}
					create_country_leader = {
						name = "Albert I"
						picture = "gfx/leaders/BEL/BEL_albert.dds"
						expire = "1935.1.1"
						ideology = autocracy
						traits = {
							trait_autocracy
						}
					}
				}
			}
		}
	}
	#play as hohenzollerns
	option = {
		name = ww1_belgium.6.a
		ai_chance = {
			factor = 100
		}
	}
	#play as albert
	option = {
		name = ww1_belgium.6.b
		ai_chance = {
			factor = 0
		}
		random_other_country = {
			limit = {
				original_tag = BEL
				has_government = democratic
			}
			change_tag_from = ROOT
		}
	}
}
