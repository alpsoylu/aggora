	id = ww1_turkey.3
	title = ww1_turkey.3.t
	desc = ww1_turkey.3.d
	picture = GFX_report_event_generic_military_parade
	trigger = {
		tag = TUR
		has_defensive_war_with = GRE
		has_global_flag = interwar_greco_turkish_war
		OR = {
			casualties > 100000
			surrender_progress > 0.2
		}
		is_in_faction = no
		NOT = {
			has_global_flag = interwar_greco_turkish_war_ended
		}
	}
	fire_only_once = yes
	mean_time_to_happen = {
		days = 0
	}
	option = {
		name = ww1_turkey.3.a		#Ask for peace
		GRE = {
			country_event = ww1_turkey.4
		}
		ai_chance = {
			factor = 10
		}
	}
	option = {
		name = ww1_turkey.3.b		#Victory or Death
		add_war_support = 0.1
		ai_chance = {
			factor = 0
		}
	}
}

			}
		}
		every_country = {
			limit = {
				OR = {
					original_tag = GRE

				}

			}
			white_peace = TUR
		}

			}

				transfer_state = PREV
			}
		}

		ai_chance = {
			factor = 100
		}
	}

		name = ww1_turkey.4.b		#ignore

			limit = {
				any_owned_state = {

	id = ww1_turkey.9
	title = ww1_turkey.9.t
	desc = ww1_turkey.9.d
	picture = GFX_report_event_generic_sign_treaty1
	is_triggered_only = yes
	option = {
		name = ww1_turkey.9.a
		hidden_effect = {
			every_state = {
				limit = {
					OR = {
						state = 341
						state = 347
						state = 756
						state = 825
					}
					OR = {
						is_owned_by = ROOT
						is_owned_by = TUR
					}
				}
				set_demilitarized_zone = yes
			}
		}
		every_owned_state = {
			limit = {
				region = 129
				is_core_of = TUR
			}
			TUR = {
				transfer_state = PREV
			}
		}
		diplomatic_relation = {
			country = TUR
			relation = non_aggression_pact
			active = yes
		}
		ai_chance = {
			factor = 100
			modifier = {
				factor = 0
				OR = {
					has_war_with = TUR
					AND = {
						original_tag = RUS
						has_global_flag = prussian_blue
					}
				}
			}
		}
	}
	option = {
		name = ww1_turkey.9.b
		trigger = {
			NOT = {
				has_global_flag = greco_turkish_war_turkey_won
			}
		}
		add_political_power = -200
		TUR = {
			add_ai_strategy = {
				type = antagonize
				id = ROOT
				value = 300
			}
		}
		ai_chance = {
			factor = 0
		}
	}
}
