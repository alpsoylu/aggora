add_namespace = ww1_finland
# Start of Regency
##################
country_event = {
	id = ww1_finland.1
	title = ww1_finland.1.t
	desc = ww1_finland.1.d
	trigger = {
		tag = FIN
		has_government = neutrality
		has_civil_war = no
		has_cosmetic_tag = FIN_senate
		has_global_flag = finnish_civil_war
	}
	fire_only_once = yes
	mean_time_to_happen = {
		days = 90
	}
	option = {
		name = ww1_finland.1.a
		drop_cosmetic_tag = yes
		set_cosmetic_tag = FIN_kingdom
		set_party_name = {
			ideology = neutrality
			long_name = FIN_regency
			name = FIN_regency
		}
		if = {
			limit = {
				has_country_leader = {
					name = "Pehr Evind Svinhufvud"
					ruling_only = yes
				}
				has_country_leader_with_trait = trait_authoritarian_democracy				
			}
			set_country_leader_ideology = oligarchism		
			swap_ruler_traits = { remove = trait_authoritarian_democracy add = trait_oligarchism }
		}
		ai_chance = {
			factor = 100
		}
	}
	option = {
		name = ww1_finland.1.b
		drop_cosmetic_tag = yes
		retire_ideology_leader = neutrality
		set_politics = {
			ruling_party = democratic
			elections_allowed = yes
		}
		ai_chance = {
			factor = 0
			modifier = {
				factor = 0
				is_historical_focus_on = yes
			}
		}
	}
}

#Fate of the regency ( Kingdom or resignation )
####################
country_event = {
	id = ww1_finland.2
	title = ww1_finland.2.t
	desc = ww1_finland.2.d
	trigger = {
		tag = FIN
		has_government = neutrality
		has_civil_war = no
		has_cosmetic_tag = FIN_kingdom
		has_global_flag = great_war_ends
	}
	fire_only_once = yes
	mean_time_to_happen = {
		days = 30
	}
	option = {
		name = ww1_finland.2.a
		trigger = {
			GER = {
				has_country_flag = great_war_winner
			}
		}
		create_country_leader = {
			name = "Fredrik Kaarle I"
			picture = "gfx/leaders/FIN/FIN_fredrik_kaarle.dds"
			expire = "1935.1.1"
			ideology = autocracy
			traits = {
				german_agent
				trait_autocracy
			}
		}
		set_party_name = {
			ideology = neutrality
			long_name = FIN_royal_house
			name = FIN_royal_house
		}
		ai_chance = {
			factor = 100
		}
	}
	option = {
		name = ww1_finland.2.b
		trigger = {
			GER = {
				has_country_flag = great_war_loser
			}
		}
		if = {
			limit = {
				NOT = {
					has_country_leader = {
						name = "Carl Gustaf Mannerheim"
						ruling_only = yes
					}			
				}
			}
			retire_ideology_leader = neutrality
		}	
		ai_chance = {
			factor = 30
		}
	}
}
