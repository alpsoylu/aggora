add_namespace = ww1_luxembourg

# DEATH OF WILLIAM IV
#######################
country_event = {
	id = ww1_luxembourg.1
	title = ww1_luxembourg.1.t
	desc = ww1_luxembourg.1.d
	picture = GFX_report_event_royal_parade
	mean_time_to_happen = {
		days = 24
	}
	trigger = {
		tag = LUX
		date > 1912.2.1
		LUX = {
			OR = {
				has_government = democratic
				has_government = neutrality
			}
			NOT = {
				has_country_flag = william_dead
			}
		}
	}
	immediate = {
		set_country_flag = william_dead
	}
	option = {
		name = ww1_luxembourg.1.a
		kill_country_leader = yes
		create_country_leader = {
			name = "Marie-Adelaide"
			picture = "gfx/leaders/LUX/LUX_marie_adelaide.dds"
			expire = "1935.1.1"
			ideology = autocracy
			traits = {
				inexperienced_monarch
				trait_autocracy
			}
		}
		ai_chance = {
			factor = 100
		}
	}
}

# ABDICATION OF MARIE-ADELAIDE
#################################
country_event = {
	id = ww1_luxembourg.2
	title = ww1_luxembourg.2.t
	desc = ww1_luxembourg.2.d
	picture = GFX_report_event_royal_parade
	trigger = {
		tag = LUX
		GER = {
			has_country_flag = great_war_loser
		}
		LUX = {
			has_country_flag = great_war_winner
			OR = {
				has_government = democratic
				has_government = neutrality
			}
			has_country_leader = {
				name = "Marie-Adelaide"
				ruling_only = no
			}
		}
	}
	fire_only_once = yes
	option = {
		name = ww1_luxembourg.2.a
		if = {
			limit = {
				has_government = neutrality
			}
			set_politics = {
				ruling_party = democratic
				elections_allowed = yes
			}
			add_popularity = {
				ideology = democratic
				popularity = 0.2
			}
		}	
		retire_ideology_leader = neutrality
		create_country_leader = {
			name = "Charlotte"
			picture = "gfx/leaders/LUX/LUX_Charlotte.dds"
			expire = "1935.1.1"
			ideology = autocracy
			traits = {
				trait_autocracy
			}
		}
		hidden_effect = {
			add_ideas = LUX_charlotte
		}
	}
}
